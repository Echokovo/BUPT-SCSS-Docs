# 查询每个制造商及其生产的最低价格的笔记本型号
```sql
select maker, min(price)
from product join laptop on product.model = laptop.model
group by maker;
```
# 查询生产的笔记本的硬盘容量不小于100GB的制造商
```sql
select maker, hd
from product join laptop on product.model = laptop.model
where hd >= 100;
```
# 查询生产最快速度的计算机的制造商
```sql
with computers as (
    select model, speed from pc
    union
    select model, speed from laptop
),
max_speed as (
    select max(speed) as speed
    from computers
)
select distinct maker
from product
join computers on product.model = computers.model
where speed = (select speed from max_speed);
```
# 用差集法查询选修了全部课程的学生姓名
```sql
select sname 
from student 
where not exists (
    select cno from course 
    except 
    select cno from sc where sno = student.sno
);
```
